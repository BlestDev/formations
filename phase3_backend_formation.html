Le rôle du backend

Le frontend (Vue) s'exécute dans le navigateur de l'utilisateur. Il ne peut pas stocker des données de façon permanente ni sécurisée, ni gérer plusieurs utilisateurs. C'est le rôle du backend.

Navigateur (Vue) Serveur (Express) Base de données │ │ │ │ GET /api/notes │ │ │ ─────────────────────────→ │ SELECT * FROM notes │ │ │ ────────────────────────→ │ │ │ ← résultats ───── │ │ ← [{id:1, titre:'...'}] ── │ │

Ce que le backend permet

  • Persistance — Les données survivent aux rechargements de page
  • Multi-utilisateurs — Chaque utilisateur a ses propres données
  • Sécurité — Les clés API et secrets restent côté serveur
  • Logique métier — Validation, calculs, envoi d'emails…

La stack qu'on va construire

Vue 3 ──── HTTP fetch() ────→ Express ────→ Supabase / SQLite (frontend) (API REST) (base de données) localhost:5173 localhost:3000
API REST en résumé Une API REST est un ensemble d'URLs (endpoints) qui répondent à des requêtes HTTP. Tu utilises déjà des API REST dans le frontend (JSONPlaceholder, Open-Meteo). Maintenant tu vas en créer une toi-même.

Middleware — les intermédiaires

Un middleware est une fonction qui s'exécute entre la requête et la réponse. Ils s'enchaînent avec next().

Middleware personnalisé — loggerJS
// Middleware : log chaque requête
const logger = (req, res, next) => {
  console.log(`${req.method} ${req.url} — ${new Date().toISOString()}`)
  next()  // ← passer au middleware/route suivant
}

app.use(logger)  // s'applique à TOUTES les routes

// Ordre d'exécution :
// Requête → logger → cors → json() → route handler → Réponse

src/routes/notes.js — CRUD complet (données en mémoire)JS
import { Router } from 'express'
const router = Router()

// Données en mémoire pour commencer (remplacées par DB au J11)
let notes = [
  { id: '1', titre: 'Première note', contenu: 'Contenu...', creeLe: new Date() }
]

// GET /api/notes — lister toutes les notes
router.get('/', (req, res) => {
  // Filtre optionnel via query param : /api/notes?recherche=vue
  const { recherche } = req.query
  let resultat = notes
  if (recherche) {
    resultat = notes.filter(n =>
      n.titre.toLowerCase().includes(recherche.toLowerCase())
    )
  }
  res.json(resultat)
})

// GET /api/notes/:id — une note
router.get('/:id', (req, res) => {
  const note = notes.find(n => n.id === req.params.id)
  if (!note) return res.status(404).json({ erreur: 'Note introuvable' })
  res.json(note)
})

// POST /api/notes — créer une note
router.post('/', (req, res) => {
  const { titre, contenu } = req.body

  // Validation basique
  if (!titre || !contenu) {
    return res.status(400).json({ erreur: 'titre et contenu requis' })
  }

  const nouvelleNote = {
    id: crypto.randomUUID(),
    titre: titre.trim(),
    contenu: contenu.trim(),
    creeLe: new Date()
  }
  notes.push(nouvelleNote)
  res.status(201).json(nouvelleNote)  // 201 = Created
})

// PUT /api/notes/:id — modifier une note
router.put('/:id', (req, res) => {
  const idx = notes.findIndex(n => n.id === req.params.id)
  if (idx === -1) return res.status(404).json({ erreur: 'Note introuvable' })

  notes[idx] = { ...notes[idx], ...req.body, id: req.params.id }
  res.json(notes[idx])
})

// DELETE /api/notes/:id — supprimer une note
router.delete('/:id', (req, res) => {
  const avant = notes.length
  notes = notes.filter(n => n.id !== req.params.id)
  if (notes.length === avant)
    return res.status(404).json({ erreur: 'Note introuvable' })
  res.status(204).send()  // 204 = No Content
})

export default router

Appeler l'API depuis Vue

src/lib/api.js (dans le projet Vue)JS
const BASE = 'http://localhost:3000/api'

export async function getNotes() {
  const res = await fetch(`${BASE}/notes`)
  if (!res.ok) throw new Error('Erreur chargement')
  return res.json()
}

export async function creerNote(donnees) {
  const res = await fetch(`${BASE}/notes`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(donnees)
  })
  if (!res.ok) throw new Error('Erreur création')
  return res.json()
}

export async function supprimerNote(id) {
  const res = await fetch(`${BASE}/notes/${id}`, { method: 'DELETE' })
  if (!res.ok) throw new Error('Erreur suppression')
}

src/routes/notes.js — avec SupabaseJS
import { Router } from 'express'
import supabase from '../db/index.js'
const router = Router()

// GET /api/notes
router.get('/', async (req, res) => {
  const { data, error } = await supabase
    .from('notes')
    .select('*')
    .order('cree_le', { ascending: false })

  if (error) return res.status(500).json({ erreur: error.message })
  res.json(data)
})

// GET /api/notes/:id
router.get('/:id', async (req, res) => {
  const { data, error } = await supabase
    .from('notes')
    .select('*')
    .eq('id', req.params.id)
    .single()

  if (error) return res.status(404).json({ erreur: 'Note introuvable' })
  res.json(data)
})

// POST /api/notes
router.post('/', async (req, res) => {
  const { titre, contenu, categorie } = req.body
  if (!titre || !contenu)
    return res.status(400).json({ erreur: 'titre et contenu requis' })

  const { data, error } = await supabase
    .from('notes')
    .insert({ titre, contenu, categorie })
    .select()
    .single()

  if (error) return res.status(500).json({ erreur: error.message })
  res.status(201).json(data)
})

// PUT /api/notes/:id
router.put('/:id', async (req, res) => {
  const { data, error } = await supabase
    .from('notes')
    .update({ ...req.body, modifie_le: new Date().toISOString() })
    .eq('id', req.params.id)
    .select()
    .single()

  if (error) return res.status(404).json({ erreur: 'Note introuvable' })
  res.json(data)
})

// DELETE /api/notes/:id
router.delete('/:id', async (req, res) => {
  const { error } = await supabase
    .from('notes')
    .delete()
    .eq('id', req.params.id)

  if (error) return res.status(404).json({ erreur: 'Note introuvable' })
  res.status(204).send()
})

export default router
src/routes/notes.js — avec SQLiteJS
import { Router } from 'express'
import db from '../db/index.js'
const router = Router()

// better-sqlite3 est SYNCHRONE (pas de async/await)

// GET /api/notes
router.get('/', (req, res) => {
  const { recherche } = req.query
  let notes
  if (recherche) {
    notes = db.prepare(
      'SELECT * FROM notes WHERE titre LIKE ? ORDER BY cree_le DESC'
    ).all(`%${recherche}%`)
  } else {
    notes = db.prepare('SELECT * FROM notes ORDER BY cree_le DESC').all()
  }
  res.json(notes)
})

// GET /api/notes/:id
router.get('/:id', (req, res) => {
  const note = db.prepare('SELECT * FROM notes WHERE id = ?')
    .get(req.params.id)
  if (!note) return res.status(404).json({ erreur: 'Note introuvable' })
  res.json(note)
})

// POST /api/notes
router.post('/', (req, res) => {
  const { titre, contenu, categorie = 'general' } = req.body
  if (!titre || !contenu)
    return res.status(400).json({ erreur: 'titre et contenu requis' })

  const id = crypto.randomUUID()
  db.prepare(`
    INSERT INTO notes (id, titre, contenu, categorie)
    VALUES (?, ?, ?, ?)
  `).run(id, titre, contenu, categorie)

  const note = db.prepare('SELECT * FROM notes WHERE id = ?').get(id)
  res.status(201).json(note)
})

// PUT /api/notes/:id
router.put('/:id', (req, res) => {
  const { titre, contenu, favori, categorie } = req.body
  const result = db.prepare(`
    UPDATE notes SET
      titre = COALESCE(?, titre),
      contenu = COALESCE(?, contenu),
      favori = COALESCE(?, favori),
      categorie = COALESCE(?, categorie),
      modifie_le = datetime('now')
    WHERE id = ?
  `).run(titre, contenu, favori, categorie, req.params.id)

  if (result.changes === 0)
    return res.status(404).json({ erreur: 'Note introuvable' })

  res.json(db.prepare('SELECT * FROM notes WHERE id = ?').get(req.params.id))
})

// DELETE /api/notes/:id
router.delete('/:id', (req, res) => {
  const result = db.prepare('DELETE FROM notes WHERE id = ?')
    .run(req.params.id)
  if (result.changes === 0)
    return res.status(404).json({ erreur: 'Note introuvable' })
  res.status(204).send()
})

export default router

Comment fonctionne l'auth par token
─── 1. Register ───────────────────────────────────────────── Client POST /register {email, password} Serveur ───────────────────────────────→ hash(password) → DB ←──── { message: "Compte créé" } ─────── ─── 2. Login ──────────────────────────────────────────────── Client POST /login {email, password} Serveur ───────────────────────────────→ vérif(password, hash) → JWT ←────────── { token: "eyJ..." } ──────── ─── 3. Requête protégée ───────────────────────────────────── Client GET /api/notes Serveur Authorization: Bearer eyJ... ───────────────────────────────→ vérifie le JWT → OK ←────────── [ liste des notes ] ────────

bcrypt — hacher les mots de passe

Ne jamais stocker un mot de passe en clair. bcrypt le transforme en une chaîne irréversible (hash). Même si la DB est compromise, les mots de passe restent sécurisés.

Terminalbash
npm install bcryptjs jsonwebtoken
bcrypt — hacher et vérifierJS
import bcrypt from 'bcryptjs'

// Hacher un mot de passe (10 = coût du hachage)
const hash = await bcrypt.hash('mon-mot-de-passe', 10)
// → "$2a$10$Xt9j0G.../..."  stocké en DB

// Vérifier un mot de passe lors du login
const valide = await bcrypt.compare('mon-mot-de-passe', hash)
// → true si le mot de passe correspond

JWT — token d'authentification

JWT — créer et vérifierJS
import jwt from 'jsonwebtoken'

const SECRET = process.env.JWT_SECRET  // clé secrète dans .env

// Créer un token (après login réussi)
const token = jwt.sign(
  { userId: '123', email: 'alice@ex.com' },  // payload
  SECRET,
  { expiresIn: '7d' }  // expire dans 7 jours
)
// → "eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9..."

// Vérifier un token (dans le middleware)
try {
  const payload = jwt.verify(token, SECRET)
  console.log(payload.userId)  // '123'
} catch (err) {
  // Token invalide ou expiré
}
JWT_SECRET dans .env Ajoute JWT_SECRET=une-longue-chaine-aleatoire-impossible-a-deviner dans ton .env. Pour générer une clé sécurisée : node -e "console.log(require('crypto').randomBytes(64).toString('hex'))"

const SECRET = process.env.JWT_SECRET

// POST /api/auth/register
router.post('/register', async (req, res) => {
  const { email, mot_de_passe, nom } = req.body

  // Validation
  if (!email || !mot_de_passe)
    return res.status(400).json({ erreur: 'email et mot_de_passe requis' })
  if (mot_de_passe.length < 8)
    return res.status(400).json({ erreur: 'Mot de passe trop court (8 min)' })

  try {
    // Hacher le mot de passe
    const hash = await bcrypt.hash(mot_de_passe, 10)
    const id = crypto.randomUUID()

    // Insérer en DB (SQLite ici — adapter pour Supabase)
    db.prepare(`
      INSERT INTO utilisateurs (id, email, mot_de_passe, nom)
      VALUES (?, ?, ?, ?)
    `).run(id, email.toLowerCase(), hash, nom || null)

    res.status(201).json({ message: 'Compte créé avec succès' })
  } catch (err) {
    // Code SQLITE_CONSTRAINT = email déjà utilisé
    if (err.code === 'SQLITE_CONSTRAINT_UNIQUE')
      return res.status(409).json({ erreur: 'Email déjà utilisé' })
    throw err
  }
})

// POST /api/auth/login
router.post('/login', async (req, res) => {
  const { email, mot_de_passe } = req.body
  if (!email || !mot_de_passe)
    return res.status(400).json({ erreur: 'Champs requis' })

  // Trouver l'utilisateur
  const utilisateur = db.prepare(
    'SELECT * FROM utilisateurs WHERE email = ?'
  ).get(email.toLowerCase())

  // Message générique (ne pas dire si c'est l'email ou le mdp)
  if (!utilisateur)
    return res.status(401).json({ erreur: 'Identifiants incorrects' })

  const mdpValide = await bcrypt.compare(mot_de_passe, utilisateur.mot_de_passe)
  if (!mdpValide)
    return res.status(401).json({ erreur: 'Identifiants incorrects' })

  // Créer le token (ne pas inclure le mot de passe!)
  const token = jwt.sign(
    { userId: utilisateur.id, email: utilisateur.email },
    SECRET,
    { expiresIn: '7d' }
  )

  res.json({
    token,
    utilisateur: {
      id: utilisateur.id,
      email: utilisateur.email,
      nom: utilisateur.nom
    }
  })
})

export default router

src/middleware/auth.jsJS
import jwt from 'jsonwebtoken'

export function requireAuth(req, res, next) {
  // Lire le header Authorization: Bearer TOKEN
  const authHeader = req.headers.authorization
  if (!authHeader || !authHeader.startsWith('Bearer ')) {
    return res.status(401).json({ erreur: 'Token manquant' })
  }

  const token = authHeader.split(' ')[1]

  try {
    const payload = jwt.verify(token, process.env.JWT_SECRET)
    req.user = payload  // disponible dans la route
    next()
  } catch {
    res.status(401).json({ erreur: 'Token invalide ou expiré' })
  }
}

src/stores/auth.js (Vue)JS
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'

export const useAuthStore = defineStore('auth', () => {
  const token = ref(localStorage.getItem('token') || null)
  const utilisateur = ref(
    JSON.parse(localStorage.getItem('user') || 'null')
  )
  const estConnecte = computed(() => !!token.value)

  async function login(email, motDePasse) {
    const res = await fetch('/api/auth/login', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ email, mot_de_passe: motDePasse })
    })
    if (!res.ok) throw new Error((await res.json()).erreur)

    const { token: t, utilisateur: u } = await res.json()
    token.value = t
    utilisateur.value = u
    localStorage.setItem('token', t)
    localStorage.setItem('user', JSON.stringify(u))
  }

  function logout() {
    token.value = null
    utilisateur.value = null
    localStorage.removeItem('token')
    localStorage.removeItem('user')
  }

  // Utiliser dans api.js :
  // headers: { Authorization: `Bearer ${token.value}` }

  return { token, utilisateur, estConnecte, login, logout }
})

Supabase fournit un système d'auth complet : email/password, OAuth (Google, GitHub…), magic link. Plus besoin de gérer bcrypt et JWT manuellement.

Dans le projet Vue — directement (sans Express)JS
import { createClient } from '@supabase/supabase-js'

const supabase = createClient(
  import.meta.env.VITE_SUPABASE_URL,
  import.meta.env.VITE_SUPABASE_KEY
)

// Créer un compte
const { data, error } = await supabase.auth.signUp({
  email: 'alice@example.com',
  password: 'motdepasse123'
})

// Connexion
const { data, error } = await supabase.auth.signInWithPassword({
  email: 'alice@example.com',
  password: 'motdepasse123'
})
// data.session.access_token = le JWT

// Déconnexion
await supabase.auth.signOut()

// Utilisateur courant
const { data: { user } } = await supabase.auth.getUser()

// Écouter les changements d'auth
supabase.auth.onAuthStateChange((event, session) => {
  if (event === 'SIGNED_IN') console.log('Connecté', session.user)
  if (event === 'SIGNED_OUT') console.log('Déconnecté')
})
