Service API dans Vue
Composable useApi · Headers JWT · Gestion erreurs centralisée
Centraliser les appels API

Plutôt que d'écrire fetch() partout avec les mêmes headers, on crée un composable qui gère ça une fois pour toutes.

src/composables/useApi.jsJS
import { useAuthStore } from '../stores/auth'

const BASE_URL = import.meta.env.VITE_API_URL || 'http://localhost:3000'

export function useApi() {
  const authStore = useAuthStore()

  async function requete(endpoint, options = {}) {
    const headers = {
      'Content-Type': 'application/json',
      ...options.headers
    }

    // Ajouter le token JWT si disponible
    if (authStore.token) {
      headers['Authorization'] = `Bearer ${authStore.token}`
    }

    const response = await fetch(`${BASE_URL}${endpoint}`, {
      ...options,
      headers,
      body: options.body ? JSON.stringify(options.body) : undefined
    })

    // Token expiré → déconnecter et rediriger
    if (response.status === 401) {
      authStore.logout()
      throw new Error('Session expirée, veuillez vous reconnecter')
    }

    if (!response.ok) {
      const erreur = await response.json().catch(() => ({}))
      throw new Error(erreur.erreur || `Erreur ${response.status}`)
    }

    // 204 No Content → pas de corps JSON
    if (response.status === 204) return null
    return response.json()
  }

  // Méthodes raccourcies
  const api = {
    get: (url) => requete(url),
    post: (url, body) => requete(url, { method: 'POST', body }),
    put: (url, body) => requete(url, { method: 'PUT', body }),
    delete: (url) => requete(url, { method: 'DELETE' })
  }

  return { api }
}

Store Pinia pour l'auth
src/stores/auth.jsJS
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { useRouter } from 'vue-router'

const BASE_URL = import.meta.env.VITE_API_URL || 'http://localhost:3000'

export const useAuthStore = defineStore('auth', () => {
  // Charger depuis localStorage au démarrage
  const token = ref(localStorage.getItem('token') || null)
  const utilisateur = ref(
    JSON.parse(localStorage.getItem('utilisateur') || 'null')
  )
  const chargement = ref(false)
  const erreur = ref(null)

  const estConnecte = computed(() => !!token.value)

  function _sauvegarder(t, u) {
    token.value = t
    utilisateur.value = u
    localStorage.setItem('token', t)
    localStorage.setItem('utilisateur', JSON.stringify(u))
  }

  async function login(email, mdp) {
    chargement.value = true
    erreur.value = null
    try {
      const res = await fetch(`${BASE_URL}/api/auth/login`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ email, mdp })
      })
      const data = await res.json()
      if (!res.ok) throw new Error(data.erreur)
      _sauvegarder(data.token, data.utilisateur)
      return true
    } catch (e) {
      erreur.value = e.message
      return false
    } finally {
      chargement.value = false
    }
  }

  async function register(nom, email, mdp) {
    chargement.value = true
    erreur.value = null
    try {
      const res = await fetch(`${BASE_URL}/api/auth/register`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ nom, email, mdp })
      })
      const data = await res.json()
      if (!res.ok) throw new Error(data.erreur)
      _sauvegarder(data.token, data.utilisateur)
      return true
    } catch (e) {
      erreur.value = e.message
      return false
    } finally {
      chargement.value = false
    }
  }

  function logout() {
    token.value = null
    utilisateur.value = null
    localStorage.removeItem('token')
    localStorage.removeItem('utilisateur')
  }

  return { token, utilisateur, chargement, erreur, estConnecte, login, register, logout }
})

Navigation Guards
Protéger les pages Vue · Rediriger si non connecté
Protéger les routes Vue

Un navigation guard s'exécute avant chaque changement de page. On l'utilise pour vérifier si l'utilisateur est connecté avant d'accéder aux pages protégées.

src/router/index.js — guards completsJS
import { createRouter, createWebHistory } from 'vue-router'
import { useAuthStore } from '../stores/auth'

const routes = [
  // Pages publiques (sans auth)
  { path: '/', name: 'accueil',
    component: () => import('../views/AccueilView.vue') },
  { path: '/login', name: 'login',
    component: () => import('../views/LoginView.vue'),
    meta: { guestOnly: true } },     // ← redirige si déjà connecté
  { path: '/register', name: 'register',
    component: () => import('../views/RegisterView.vue'),
    meta: { guestOnly: true } },

  // Pages protégées (auth requise)
  { path: '/notes', name: 'notes',
    component: () => import('../views/NotesView.vue'),
    meta: { requiresAuth: true } },
  { path: '/notes/nouvelle', name: 'nouvelle-note',
    component: () => import('../views/NouvelleNoteView.vue'),
    meta: { requiresAuth: true } },
  { path: '/notes/:id', name: 'note-detail',
    component: () => import('../views/NoteDetailView.vue'),
    meta: { requiresAuth: true } },
  { path: '/profil', name: 'profil',
    component: () => import('../views/ProfilView.vue'),
    meta: { requiresAuth: true } },

  { path: '/:pathMatch(.*)*', name: 'not-found',
    component: () => import('../views/NotFoundView.vue') }
]

const router = createRouter({
  history: createWebHistory(),
  routes
})

// Guard global — s'exécute avant CHAQUE navigation
router.beforeEach((to) => {
  const auth = useAuthStore()

  // Page protégée + pas connecté → login
  if (to.meta.requiresAuth && !auth.estConnecte) {
    return { name: 'login', query: { redirect: to.fullPath } }
  }

  // Page guest only + déjà connecté → notes
  if (to.meta.guestOnly && auth.estConnecte) {
    return { name: 'notes' }
  }
})

export default router

Store notes branché sur l'API

On adapte le store Pinia de la Phase 2 pour qu'il appelle l'API au lieu de gérer localStorage.

src/stores/notes.js — version APIJS
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { useApi } from '../composables/useApi'

export const useNotesStore = defineStore('notes', () => {
  const { api } = useApi()

  const notes = ref([])
  const chargement = ref(false)
  const erreur = ref(null)
  const recherche = ref('')

  const notesFiltrees = computed(() =>
    notes.value.filter(n =>
      n.titre.toLowerCase().includes(recherche.value.toLowerCase())
    )
  )

  // Charger toutes les notes depuis l'API
  async function charger() {
    chargement.value = true
    erreur.value = null
    try {
      notes.value = await api.get('/api/notes')
    } catch (e) {
      erreur.value = e.message
    } finally {
      chargement.value = false
    }
  }

  async function creer(donnees) {
    const nouvelleNote = await api.post('/api/notes', donnees)
    notes.value.unshift(nouvelleNote)   // ajouter en début de liste
    return nouvelleNote
  }

  async function mettreAJour(id, donnees) {
    const noteMAJ = await api.put(`/api/notes/${id}`, donnees)
    const idx = notes.value.findIndex(n => n.id === id)
    if (idx >= 0) notes.value[idx] = noteMAJ
    return noteMAJ
  }

  async function supprimer(id) {
    await api.delete(`/api/notes/${id}`)
    notes.value = notes.value.filter(n => n.id !== id)
  }

  return {
    notes, chargement, erreur, recherche,
    notesFiltrees, charger, creer, mettreAJour, supprimer
  }
})

Gérer les environnements

En local tu utilises localhost. En production, tu as de vraies URLs. Les variables d'environnement permettent de changer ces valeurs sans modifier le code.

Frontend Vue (Vite)

Fichiers .env dans le projet Vuebash
# .env — local (gitignore)
VITE_API_URL=http://localhost:3000

# .env.production — production (peut être commité)
VITE_API_URL=https://mon-api.railway.app

# Utilisation dans le code Vue :
# import.meta.env.VITE_API_URL
Seules les variables VITE_ sont exposées au navigateur Vite inclut uniquement les variables préfixées VITE_ dans le bundle. Ne mets jamais de secrets (clés privées, mots de passe DB) dans le .env du frontend — ils seraient visibles par tous.

Backend Express

Variables backend — configurées sur la plateformebash

JWT_SECRET=mon-secret-local-tres-long
SUPABASE_URL=https://xxxxx.supabase.co
SUPABASE_SERVICE_KEY=eyJh...

# En production (Railway, Render, etc.) :
# Ces variables sont définies dans le dashboard de la plateforme
# → Elles ne sont JAMAIS dans le code source

JWT_SECRET=...      # générer avec : openssl rand -base64 32

Debugger en production
Les erreurs courantes et comment les résoudre
Problèmes fréquents & solutions
Erreur CORSAccess-Control-Allow-Origin manquant

Symptôme : La console Vue affiche CORS error, le réseau montre une requête bloquée.

Causes & solutions :

  • Le FRONTEND_URL dans Railway ne correspond pas exactement à l'URL Vercel (attention au slash final)
  • Tu as oublié app.use(cors(...)) avant les routes dans server.js
  • Vérifier les logs Railway → est-ce que le serveur démarre sans erreur ?
Erreur 401Token invalide ou manquant

Symptôme : Les requêtes API retournent 401 même après connexion.

  • Ouvrir les DevTools → Application → localStorage → vérifier que token est présent
  • Dans useApi.js, vérifier que le header Authorization: Bearer xxx est bien envoyé (onglet Network)
  • Le JWT_SECRET en production est différent du local ? Reconnecter.
Page blancheBuild Vue qui ne fonctionne pas
  • Ouvrir la console du navigateur → erreur JS visible ?
  • Vérifier que VITE_API_URL est configurée dans Vercel (pas seulement en local)
  • Vérifier vercel.json pour le routing SPA
  • Tenter un redéploiement forcé dans Vercel
Railway crashLe backend ne démarre pas
  • Consulter les logs dans Railway → message d'erreur exact
  • Vérifier que package.json a un script "start"
  • Toutes les variables d'env sont configurées ? (JWT_SECRET, SUPABASE_URL…)
  • Tester npm start en local avec les mêmes variables
